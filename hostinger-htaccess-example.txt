# .htaccess dla Hostinger - umieść w folderze /public_html/downloads/
# Ten plik umożliwia bezpośredni dostęp do plików Divi ZIP

# Włącz moduł headers (jeśli dostępny)
<IfModule mod_headers.c>
    # Zezwól na CORS
    Header set Access-Control-Allow-Origin "*"
</IfModule>

# Konfiguracja dla wszystkich plików ZIP w tym katalogu
<FilesMatch "\.(zip)$">
    # Zezwól na dostęp do pliku
    Require all granted

    # Ustaw prawidłowy typ MIME
    ForceType application/zip

    # Opcjonalnie: Wymuszaj download zamiast wyświetlania w przeglądarce
    # Header set Content-Disposition "attachment"
</FilesMatch>

# Specyficzna konfiguracja dla plików Divi (opcjonalne)
<Files "Divi*.zip">
    Require all granted
    ForceType application/zip
</Files>

# Opcje dla katalogu
Options -Indexes
DirectoryIndex disabled
